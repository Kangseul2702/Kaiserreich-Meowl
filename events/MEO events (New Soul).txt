country_event = {
	id = meowl_events.2
	title = meowl_events.2.t
	desc = meowl_events.2.d
	picture = GFX_report_event_Meowl_events_01

	trigger = {
		is_american_tag = yes
		OR = {
			country_exists = ACC
			country_exists = WCA
			country_exists = APG
		}
		NOT = { country_exists = MEO }
	}

	fire_only_once = yes

	option = {
		log = "[GetLogInfo]: event meowl_events.2 option meowl_events.2.a"
		name = meowl_events.2.a
		386 = { set_demilitarized_zone = yes }
	}

	option = {
		log = "[GetLogInfo]: event meowl_events.2 option meowl_events.2.b"
		name = meowl_events.2.b
		ai_chance = { base = 0 }
		custom_effect_tooltip = tooltip_no_meowl
		set_global_flag = MEO_NoMeowl
	}
}

country_event = {
	id = meowl_events.3
	title = meowl_events.3.t
	desc = meowl_events.3.d
	picture = GFX_report_event_Meowl_events_01

	trigger = {
		is_american_tag = yes
		has_global_flag = USA_civil_war_begun_flag
		has_war = yes
		is_ai = no
		NOT = {
			country_exists = MEO
			has_global_flag = MEO_NoMeowl
		}
	}

	immediate = {
		ACC = { save_event_target_as = acw_country }
		NEE = { save_event_target_as = acw_country }
		APG = { save_event_target_as = acw_country }
		USA = { save_event_target_as = acw_country }
	}

	fire_only_once = yes

	option = {
		log = "[GetLogInfo]: event meowl_events.3 option meowl_events.3.a"
		name = meowl_events.3.a
		ACW_release_MEO = yes
	}

	option = {
		log = "[GetLogInfo]: event meowl_events.3 option meowl_events.3.b"
		name = meowl_events.3.b
		ACW_release_MEO = yes
		MEO = {
			change_tag_from = ROOT
			open_intro_screen = yes
		}
	}
}
